## operator

1.  operator ++

```js
const a = 0
const b = a++
const c = ++a
```

2.  operator ,

```js
const a = 1
const b = (a++, 4)
```

3.  operator \\&\\&

```js
const a = 1
a === 1 && do() // 等同于 if(a === 1){ do() }
```

4.  operator ||

```js
const a = 1
a === 1 || do() // 等同于 if(a !== 1){ do() }
```

5.  operator \~~

```js
const num = ~~ 1.23 // 1
```

6.  operator |

```js
const num = 1.23 | 0 // 1
```

7.  operator >>

```js
const num = 1.23 >> 0 // 1
```

8.  void

```js
void function() {console.log('iife')}() // 'iife'
function() {console.log('iife')}() // Uncaught SyntaxError: Function statements require a function name
(function() {console.log('iife')})() // 'iife'
```
